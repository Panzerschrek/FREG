28  static int c;

30  case KEY_UP:
31  case 'u': emit InputReceived(MOVE, NORTH); break;
32  case KEY_DOWN:
33  case 'D': emit InputReceived(MOVE, SOUTH); break;
34  case KEY_RIGHT:
35  case 'r': emit InputReceived(MOVE, EAST ); break;

38  case '>': emit InputReceived(TURN_RIGHT, HERE); break;
39  case '<': emit InputReceived(TURN_LEFT, HERE); break;
40  case ' ': emit InputReceived(JUMP, HERE); break;
41  case KEY_NPAGE:
42  case 'v': emit InputReceived(TURN, DOWN); break;
43  case KEY_PPAGE:
44  case 'p': emit InputReceived(TURN, UP); break;
45  case 'i': emit InputReceived(OPEN_INVENTORY, HERE); break;

49  case 'd': emit InputReceived(DROP, HERE); break;

53  case 'I': emit InputReceived(INSCRIBE, HERE); break;
54  case KEY_BACKSPACE:
55  case 'B': emit InputReceived(DAMAGE, HERE); break;
56  case KEY_HOME:
57  case 'H': emit InputReceived(BUILD, HERE); break;
58  case 'L': emit RePrintReceived(); break;
59  case 'Q': emit ExitReceived(); break;
60  default: break;
61  }
