47 int Block::Damage(

49  const damage_kinds dmg_kind=CRUSH) {

52 
53  switch (sub) {
54  case GLASS: return durability=0;
55  case MOSS_STONE:
56  case STONE:

58  durability-=2*dmg :

60  case GREENERY: case ROSE: case HAZELNUT: case WOOD:
61  return (CUT==dmg_kind) ?
62  durability-=2*dmg :
63  durability-=dmg;
64  case SAND: case SOIL:
65  return (DIG==dmg_kind) ?
66  durability-=2*dmg :
67  durability-=dmg;
68  case A_MEAT: case H_MEAT:
69  return (THRUST==dmg_kind) ?
70  durability-=2*dmg :
71  durability-=dmg;
72  case DIFFERENT: case AIR: case SKY: case SUN_MOON: case WATER:
73  case NULLSTONE: case STAR:
74  return durability;
75  default: return durability-=dmg;

328 bool Rabbit::Act() {
329  World * const world=GetWorld();
330  float for_north=0, for_west=0;
331  ushort x, y, z;
332  int kind;
333  for (x=x_self-6; x<=x_self+6; ++x)
334  for (y=y_self-6; y<=y_self+6; ++y)
335  for (z=z_self-6; z<=z_self+6; ++z)
336  if ( world->InBounds(x, y, z) ) {
337  kind=world->Kind(x, y, z);
338  if ( (GRASS==kind || RABBIT==kind || DWARF==kind) &&
339  world->DirectlyVisible(x_self, y_self, z_self, x, y, z) ) {
340  if ( y!=y_self )
341  for_north+=Attractive(kind)/(y_self-y);
342  if ( x!=x_self )
343  for_west +=Attractive(kind)/(x_self-x);
344  }
345  }
346 
347  if ( abs(for_north)>1 || abs(for_west)>1 ) {
348  SetDir( ( abs(for_north)>abs(for_west) ) ?
349  ( ( for_north>0 ) ? NORTH : SOUTH ) :
350  ( ( for_west >0 ) ? WEST : EAST ) );
351  if ( rand()%2 )
352  world->Move(x_self, y_self, z_self, direction);
353  else
354  world->Jump(x_self, y_self, z_self);
355  } else switch ( rand()%60 ) {
356  case 0: world->Move(x_self, y_self, z_self, NORTH); break;
357  case 1: world->Move(x_self, y_self, z_self, SOUTH); break;
358  case 2: world->Move(x_self, y_self, z_self, EAST); break;
359  case 3: world->Move(x_self, y_self, z_self, WEST); break;
360  }
361 
362  if ( seconds_in_day*time_steps_in_sec/2>satiation ) {
363  for (x=x_self-1; x<=x_self+1; ++x)
364  for (y=y_self-1; y<=y_self+1; ++y)
365  for (z=z_self-1; z<=z_self+1; ++z)
366  if ( GREENERY==world->Sub(x, y, z) ) {
367  world->Eat(x_self, y_self, z_self, x, y, z);
368  return Animal::Act();
369  }
370  }
371 
372  return Animal::Act();
373 }
