315  public:
316  QString & InvFullName(QString & str, const ushort i) const {
317  return str=( inventory[i] ) ?
318  inventory[i]->FullName(str) :
319  "";
320  }
321  char * NumStr(char * const str, const ushort i) const {
322  const ushort n=Number(i);
323  if ( 1>=n )
324  str[0]='\0';
325  else
326  sprintf(str, " (%hdx)", n);
327  return str;
328  }
329  float GetInvWeight(const ushort i) const {
330  return ( inventory[i] ) ?
331  inventory[i]->Weight()*Number(i) :
332  0;
333  }
334  int GetInvSub(const ushort i) const {
335  return ( inventory[i] ) ?
336  inventory[i]->Sub() :
337  AIR;
338  }
339  int GetInvKind(const ushort i) const {
340  return ( inventory[i] ) ?
341  inventory[i]->Kind() :
342  BLOCK;
343  }
344  float InvWeightAll() const {
345  float sum=0;
346  for (ushort i=0; i<inventory_size; ++i)
347  sum+=GetInvWeight(i);
348  return sum;
349  }
350  int Number(const ushort i) const {
351  return inventory_num[i];
352  }
353 
354  virtual QString & FullName(QString&) const=0;
355  virtual int Kind() const=0;
356  virtual int Sub() const=0;
357  virtual bool Access() const { return true; }
358 
359  virtual Inventory * HasInventory() { return this; }
360  usage_types Use() { return OPEN; }
361 
362  Block * Drop(const ushort n) {
363  if ( inventory_size<=n ||
364  !inventory_num[n])
365  return 0;
366 
367  Block * const temp=new Block(*inventory[n]);
368  --inventory_num[n];
369  if ( !inventory_num[n] ) {
370  if ( !inventory[n]->Normal() )
371  delete inventory[n];
372  inventory[n]=0;
373  }
374  return temp;
375  }
376  virtual int Get(Block * const block) {
377  if ( !block )
378  return 1;
379 
380  for (ushort i=0; i<inventory_size; ++i) {
381  if ( !inventory[i] ) {
382  inventory[i]=block;
383  inventory_num[i]=1;
384  return 1;
385  }
386  if ( *block==*inventory[i] &&
387  inventory_num[i]<max_stack_size ) {
388  ++inventory_num[i];
389  if ( !block->Normal() )
390  delete block;
391  return 1;
392  }
393  }
394  return 0;
395  }
396  void GetAll(Inventory * const from) {
397  if ( !from || !from->Access() )
398  return;
399 
400  for (ushort i=0; i<inventory_size; ++i)
401  while ( from->inventory_num[i] ) {
402  Block * const temp=from->Drop(i);
403  if ( !Get(temp) ) {
404  from->Get(temp);
405  return;
406  }
407  }
408  }
409  virtual void SaveAttributes(QDataStream & out) const {
410  for (ushort i=0; i<inventory_size; ++i) {
411  out << inventory_num[i];
412  if ( inventory[i] )
413  inventory[i]->SaveToFile(out);
414  }
415  }
416 
417  Inventory(Shred * const sh)
418  :
419  inShred(sh)
420  {
421  for (ushort i=0; i<inventory_size; ++i) {
422  inventory[i]=0;
423  inventory_num[i]=0;
424  }
425  }
426  Inventory(Shred * const,
427  QDataStream & str);
428  ~Inventory() {
429  for (ushort i=0; i<inventory_size; ++i)
430  if ( inventory[i] && !inventory[i]->Normal() )
431  delete inventory[i];
432  }
