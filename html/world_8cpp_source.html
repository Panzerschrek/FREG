62 int World::MakeDir(
63  const ushort x_center,
64  const ushort y_center,
65  const ushort x_target,
66  const ushort y_target) const
67 {
68  //if (x_center==x_target && y_center==y_target) return HERE;
69  if ( abs(x_center-x_target)<=1 && abs(y_center-y_target)<=1 )
70  return HERE;
71  const float x=x_target-x_center;
72  const float y=y_target-y_center;
73  if ( y <= 3*x && y <= -3*x ) return NORTH;
74  else if ( y > -3*x && y < -x/3 ) return NORTH_EAST;
75  else if ( y >= -x/3 && y <= x/3 ) return EAST;
76  else if ( y > x/3 && y <3 *x ) return SOUTH_EAST;
77  else if ( y >= 3*x && y >= -3*x ) return SOUTH;
78  else if ( y <- 3*x && y >- x/3 ) return SOUTH_WEST;
79  else if ( y <=- x/3 && y >= x/3 ) return WEST;
80  else return NORTH_WEST;
81 }

205  static const ushort start=numShreds/2-numActiveShreds/2;
206  static const ushort end=start+numActiveShreds;
207  if ( time_steps_in_sec>timeStep ) {
208  ++timeStep;
209  for (ushort i=start; i<end; ++i)
210  for (ushort j=start; j<end; ++j)
211  shreds[i+j*NumShreds()]->PhysEvents();
212 
213  Unlock();

215  }
216 
217  timeStep=0;
218  ++time;

220  //sun/moon moving

230  emit Updated(sun_moon_x, y, height-1);
231  }

233  switch ( TimeOfDay() ) {
234  case end_of_evening:

236  }
237

242  float x_from,
243  float y_from,
244  float z_from,
245  const ushort x_to,
246  const ushort y_to,
247  const ushort z_to) const
248 {

250  return true;
251

253  const ushort ydif=abs(y_to-(int)y_from);
254  const ushort zdif=abs(z_to-(int)z_from);
255  ushort max=(zdif > ydif) ?
256  zdif :

258  if ( xdif > max )
259  max=xdif;
260 
261  const float x_step=(float)(x_to-x_from)/max;
262  const float y_step=(float)(y_to-y_from)/max;
263  const float z_step=(float)(z_to-z_from)/max;
264 
265  for (ushort i=1; i<max; ++i)
266  if ( !Transparent(
267  nearbyint(x_from+=x_step),
268  nearbyint(y_from+=y_step),
269  nearbyint(z_from+=z_step)) )
270  return false;
271  return true;

291 int World::Move(
292  const ushort i,
293  const ushort j,
294  const ushort k,
295  const int dir,
296  const ushort stop) //see default in "world.h"
297 {
298  if ( !InBounds(i, j, k) )
299  return 0;
300 
301  ushort newi, newj, newk;
302  if ( NOT_MOVABLE==Movable(i, j, k) ||
303  Focus(i, j, k, newi, newj, newk, dir) )
304  return 0;
305 
306  Block * const block=GetBlock(i, j, k);
307  if ( DESTROY==(block->BeforeMove(dir)) ) {
308  delete block;
309  SetBlock(NewNormal(AIR), i, j, k);
310  return 1;
311  }
312 
313  if ( !stop || (ENVIRONMENT==Movable(i, j, k) &&
314  Equal(block, GetBlock(newi, newj, newk))) )
315  return 0;
316 
317  short numberMoves=0;
318  if ( ENVIRONMENT!=Movable(newi, newj, newk) &&
319  !(numberMoves=Move(newi, newj, newk, dir, stop-1)) )
320  return 0;
321 
322  SetBlock(GetBlock(newi, newj, newk), i, j, k);
323  SetBlock(block, newi, newj, newk);
324 
325  ReEnlighten(newi, newj, newk);
326  ReEnlighten(i, j, k);
327 
328  if ( GetBlock(i, j, k) )
329  GetBlock(i, j, k)->Move( Anti(dir) );
330  GetBlock(newi, newj, newk)->Move(dir);
331 
332  if ( Weight(newi, newj, newk) ) {
333  if ( Weight(newi, newj, newk) >
334  Weight(newi, newj, newk-1) )
335  numberMoves+=Move(newi, newj, newk, DOWN, stop-1);
336  else if ( Weight(newi, newj, newk) <
337  Weight(newi, newj, newk+1) )
338  numberMoves+=Move(newi, newj, newk, UP, stop-1);
339  }
340 
341  return ++numberMoves;
342 }

344 void World::Jump(
345  const ushort i,
346  const ushort j,
347  ushort k)
348 {
349  Block * const to_move=GetBlock(i, j, k);

358  to_move->SetWeight();

360  Move(i, j, k, DOWN);
361  } else
362  to_move->SetWeight();
